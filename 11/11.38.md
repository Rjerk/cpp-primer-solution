#include "head.h"

class TextTrans {
public:
    void word_transfrom(ifstream&, ifstream&);
    unordered_map<string, string> build_map(ifstream&);
    const string& transform(const string&, const unordered_map<string, string>&);
};

void TextTrans::word_transfrom(ifstream& map_file, ifstream& input)
{
    unordered_map<string, string> map_format = build_map(map_file);
    string line;
    while (getline(input, line)) {
        istringstream wordstream(line);
        string word;
        while (wordstream >> word) {
            cout << transform(word, map_format) << " ";
        }
        cout << endl;
    }
}

unordered_map<string, string> TextTrans::build_map(ifstream& mapfile)
{
    unordered_map<string, string> map_format;
    string raw_word;
    string new_word;
    while (mapfile >> raw_word && getline(mapfile, new_word)) {
        map_format[raw_word] = new_word;
    }
    return map_format;
}

const string& TextTrans::transform(const string& word, const unordered_map<string, string>& map_format)
{
    auto found = map_format.find(word);
    if (found != map_format.cend())
        return found->second;
    else
        return word;
}

int main()
{
    TextTrans t;
    ifstream map_file("map_file.txt");
    ifstream to_trans("to_trans.txt");
    t.word_transfrom(map_file, to_trans);
    return 0;
}
