class HasPtr {
public:
    HasPtr(const string& s = string()) : ps(new string(s)), i(0) { }
    HasPtr(const HasPtr& ptr): ps(new string(*ptr.ps)), i(ptr.i) { }
    HasPtr& operator=(const HasPtr& ptr);
    ~HasPtr() { }
private:
    string* ps;
    int i;
};

HasPtr& HasPtr::operator=(const HasPtr& ptr)
{
    delete ps;
    ps = new string(*ptr.ps);
    i = ptr.i;
