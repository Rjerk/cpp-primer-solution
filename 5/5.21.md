#include "head.h"

int main()
{
    string word;
    string pre_word = "";
    unsigned times = 0;
    while (cin >> word) {
        if (word == pre_word && isupper(word[0])) {
            ++times;
            cout << word << endl;
            break;
        }
        else {
            times = 1;
        }
        pre_word = word;
    }

    if (times == 1) {
        cout << "Words has not continuity" << endl;
    }
    return 0;
}
